<DEFAULT> SKIP : {
" "
| "\t"
| "\n"
| "\r"
| "\f"
}

   
<DEFAULT> SPECIAL : {
<SINGLE_LINE_COMMENT: "//" (~["\n","\r"])* ("\n" | "\r" | "\r\n")>
| <FORMAL_COMMENT: "/**" (~["*"])* "*" ("*" | ~["*","/"] (~["*"])* "*")* "/">
| <MULTI_LINE_COMMENT: "/*" (~["*"])* "*" ("*" | ~["*","/"] (~["*"])* "*")* "/">
}

   
<DEFAULT> TOKEN : {
<LPAREN: "(">
| <RPAREN: ")">
| <SEMICOLON: ";">
| <ASSIGN: "=">
| <PLUS: "+">
| <MINUS: "-">
| <TIMES: "*">
| <PRINT: "print">
}

   
<DEFAULT> TOKEN : {
<INTEGER_LITERAL: ["1"-"9"] (["0"-"9"])* | "0">
}

   
<DEFAULT> TOKEN : {
<IDENTIFIER: <LETTER> (<LETTER> | <DIGIT>)*>
| <#LETTER: ["$","A"-"Z","_","a"-"z","\u00c0"-"\u00d6","\u00d8"-"\u00f6","\u00f8"-"\u00ff","\u0100"-"\u1fff","\u3040"-"\u318f","\u3300"-"\u337f","\u3400"-"\u3d2d","\u4e00"-"\u9fff","\uf900"-"\ufaff"]>

}

<COMPSTAT: <TLETTER> <FLETTER>>

}

/************************************
 * The MiniExpr Grammar Starts Here *
 ************************************/

// BNF
Goal ::= ( PStatement )* <EOF>
PStatement ::= Statementt
| Statement
Statement ::= Statement1 Statement2 ";"
Statementt ::= Statement1 Statement3 Statement2 ";"
Statement3 ::= Compstat
Statement1 ::= Identifier
Statement2 ::= Identifier
AssignmentStatement ::= Identifier "=" Expression ";"
PrintStatment ::= <PRINT> Expression ";"
Expression ::= PlusExpression
| MinusExpression
| TimesExpression
| PrimaryExpression
PlusExpression ::= PrimaryExpression <PLUS> PrimaryExpression
MinusExpression ::= PrimaryExpression <MINUS> PrimaryExpression
TimesExpression ::= PrimaryExpression <TIMES> PrimaryExpression
PrimaryExpression ::= IntegerLiteral
| Identifier
| BracketExpression
IntegerLiteral ::= <INTEGER_LITERAL>
Identifier ::= <IDENTIFIER>
Compstat ::= <COMPSTAT>
BracketExpression ::= <LPAREN> Expression ")"
